export const realtorTrustedPros = pgTable(
  'realtor_trusted_pros', {
  id: serial('id').primaryKey(),
  realtorId: integer('realtor_id')
    .notNull().references(() => users.id),
  proId: integer('pro_id')
    .notNull().references(() => users.id),
  note: text('note'),
  sortOrder: integer('sort_order').default(0),
  createdAt: timestamp('created_at')
    .defaultNow().notNull(),
});

Exporta desde lib/db/src/schema/index.ts

PASO 2 — Backend
Crea artifacts/api-server/src/routes/realtor.ts

Endpoints:

GET /api/realtor/profile
  → Perfil del realtor autenticado
  → Devuelve: id, name, email, phone,
    brokerage, licenseNumber, 
    profilePhoto, serviceCity, 
    createdAt

PATCH /api/realtor/profile
  → Actualizar perfil del realtor
  → Body: { brokerage?, licenseNumber?,
    phone?, serviceCity?, bio? }

GET /api/realtor/trusted-pros
  → Lista de pros recomendados
  → Devuelve: proId, name, category,
    serviceCity, profilePhoto,
    rating promedio, note

POST /api/realtor/trusted-pros
  → Agregar un pro a la lista
  → Body: { proId: number, note?: string }

DELETE /api/realtor/trusted-pros/:proId
  → Quitar un pro de la lista

GET /api/realtor/public/:realtorId
  → Perfil público del realtor
  → Sin autenticación requerida
  → Devuelve: perfil completo + 
    lista de trusted pros

Registra en routes/index.ts:
import realtorRouter from "./realtor";
router.use(realtorRouter);
