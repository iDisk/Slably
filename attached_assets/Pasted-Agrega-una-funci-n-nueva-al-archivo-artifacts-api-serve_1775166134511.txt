export async function extractReceiptData(
  imageBuffer: Buffer,
  mimeType: string,
): Promise<{
  vendor: string | null;
  amount: number | null;
  date: string | null;
  description: string | null;
  category: 'materials' | 'labor' | 'equipment' | 'permits' | 'other' | null;
  payment_method: 'cash' | 'card' | 'transfer' | 'check' | null;
  confidence: number;
}> {
  const openai = getOpenAI();
  const base64 = imageBuffer.toString("base64");
  const dataUrl = `data:${mimeType};base64,${base64}`;

  const completion = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    temperature: 0,
    max_tokens: 500,
    messages: [
      {
        role: "system",
        content: `Eres un asistente que extrae datos de tickets y recibos de construcción.
Analiza la imagen y devuelve SOLO un objeto JSON válido sin markdown ni backticks con estos campos:
{
  "vendor": "nombre del proveedor o tienda (ej. Home Depot, 84 Lumber, etc.) o null",
  "amount": número total en dólares (solo el número, sin símbolos) o null,
  "date": "fecha en formato YYYY-MM-DD o null",
  "description": "descripción breve de los items comprados o null",
  "category": "materials|labor|equipment|permits|other o null",
  "payment_method": "cash|card|transfer|check o null",
  "confidence": número del 0 al 100 indicando qué tan legible es el ticket
}
Si el ticket no es legible o la imagen no es un recibo, devuelve confidence: 0.
Responde SOLO con el JSON.`,
      },
      {
        role: "user",
        content: [
          {
            type: "image_url",
            image_url: { url: dataUrl, detail: "high" },
          },
          {
            type: "text",
            text: "Extrae los datos de este ticket/recibo de construcción.",
          },
        ],
      },
    ],
  });

  const raw = completion.choices[0]?.message?.content ?? "{}";
  try {
    const parsed = JSON.parse(raw);
    return {
      vendor:         parsed.vendor         ?? null,
      amount:         parsed.amount         ?? null,
      date:           parsed.date           ?? null,
      description:    parsed.description    ?? null,
      category:       parsed.category       ?? null,
      payment_method: parsed.payment_method ?? null,
      confidence:     typeof parsed.confidence === "number" 
                        ? parsed.confidence : 50,
    };
  } catch {
    return {
      vendor: null, amount: null, date: null,
      description: null, category: null,
      payment_method: null, confidence: 0,
    };
  }
}
